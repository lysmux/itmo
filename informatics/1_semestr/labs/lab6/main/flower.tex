\documentclass{article}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usepackage{amsmath}
\usepackage{tikz}
\usepgfplotslibrary{polar}

\begin{document}
	
\begin{tikzpicture}
	\begin{axis}[
		xlabel={$x$},
		ylabel={$y$},
		axis equal,
		grid=both,
		xmin=-1, xmax=1,
		ymin=-1, ymax=1,
		]
		\addplot [
		domain=0:360,
		samples=200,
		color=red
		]
		({sin(6*x) * cos(x)}, {sin(6*x) * sin(x)});
	\end{axis}
\end{tikzpicture}	
	
\begin{tikzpicture}
	\begin{polaraxis}[]
		\addplot[
		domain=0:360,
		samples=200,
		color=red
		]
		{sin(6*x)};
	\end{polaraxis}
\end{tikzpicture}
